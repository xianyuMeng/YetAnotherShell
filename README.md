 Grammer
  
  Simple cmd
    
    ;word-list
    
    ;

    : simple-cmd RT WORD LT WORD
    
    | simple-cmd LT WORD RT WORD
    
    | simple-cmd LT WORD
    
    | simple-cmd RT WORD
    
    | simple-cmd
    
    ;
  
  Pipe cmd
    
    : redir-cmd
    
    | redir-cmd PIPE pipe-cmd
